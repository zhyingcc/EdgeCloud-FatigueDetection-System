41 struct StableElement {
45  using StableElementT = StableElement<T, Compare>;
46 
48  StableElement(T&& o, std::size_t c) : object_(std::move(o)), order_(c) {}
49 
51  StableElement(const T& o, std::size_t c) : object_(o), order_(c) {}
52 
54  virtual ~StableElement() = default;
55 
57  explicit operator T() { return object_; }
58 
60  bool operator<(const StableElementT& rhs) const {
61  /* if element priority is same, compare with order */
62  if (comp_(object_, rhs.object_) == false &&
63  comp_(rhs.object_, object_) == false) {
64  return order_ >= rhs.order_;
65  }
66 
67  /* compare two objects */
68  return comp_(object_, rhs.object_);
69  }
70 
72  T object_;
74  std::size_t order_;
76  Compare comp_;
77 };

80 template <typename T, typename Compare = std::less<T>>
81 class StablePriorityQueue
82  : public std::priority_queue<StableElement<T, Compare>,
83  std::vector<StableElement<T, Compare>>,
84  std::less<StableElement<T, Compare>>> {
85  using stableT = StableElement<T, Compare>;
86  using std::priority_queue<
87  stableT, std::vector<StableElement<T, Compare>>,
88  std::less<StableElement<T, Compare>>>::priority_queue;
89 
90  public:
92  StablePriorityQueue() = default;
94  virtual ~StablePriorityQueue() = default;
96  const T& front() { return this->c.front().object_; }
98  const T& top() { return this->c.front().object_; }
100  void push(const T& value) {
101  /* push back and increase order */
102  this->c.push_back(stableT(value, counter_++));
103  std::push_heap(this->c.begin(), this->c.end(), this->comp);
104  }
105 
107  void push(T&& value) {
108  this->c.push_back(stableT(std::move(value), counter_++));
109  std::push_heap(this->c.begin(), this->c.end(), this->comp);
110  }
111 
113  template <class... Args>
114  void emplace(Args&&... args) {
115  /* emplace element */
116  this->c.emplace_back(T(std::forward<Args>(args)...), counter_++);
117  std::push_heap(this->c.begin(), this->c.end(), this->comp);
118  }
119 
121  void pop() {
122  /* pop element */
123  std::pop_heap(this->c.begin(), this->c.end(), this->comp);
124  this->c.pop_back();
125  /* if queue is empty, reset counter, this will avoid counter overflow */
126  if (this->empty()) {
127  counter_ = 0;
128  }
129  }
130 
131  protected:
136  std::size_t counter_ = 0;
137 };

493  bool Wait(std::condition_variable& cond, std::unique_lock<std::mutex>& lock,
494  int timeout, const std::function<bool()> &wait_cond) {
495  bool ret = false;
496  auto cond_func = [&]() { return need_wakeup_ || wait_cond(); };
497 
498  waiter_number_++;
499  if (timeout > 0) {
500  /* if timeout is set, wait for timeout and return false */
501  ret = cond.wait_for(lock, std::chrono::milliseconds(timeout), cond_func);
502  if (ret == false) {
503  errno = ETIMEDOUT;
504  }
505  } else if (timeout == 0) {
506  /* if wait forever, do wait */
507  cond.wait(lock, cond_func);
508  ret = true;
509  } else {
510  /* do not wait */
511  ret = cond_func();
512  }
513 
514  waiter_number_--;
515  /* if wake by Wakeup */
516  if (need_wakeup_) {
517  /* if all waiters have been woken up */
518  if (waiter_number_ == 0) {
519  need_wakeup_ = false;
520  }
521  errno = EINTR;
522  ret = false;
523  }
524 
525  /* wakeup */
526  return ret;

605  size_t PushQueue(std::unique_lock<std::mutex>& lock, Sequence* elems,
606  int timeout = 0, size_t expect_space = 1) {
607  size_t push_num = 0;
608 
609  /* shutdown or queue has enough space */
610  auto wait_check = [&]() {
611  return shutdown_ == true || queue_.size() == 0 ||
612  capacity_ >= expect_space + queue_.size();
613  };
614 
615  /* check if queue has space, or wait. */
616  if (Wait(not_full_, lock, timeout, wait_check) == false) {
617  return false;
618  }
619 
620  for (auto it = elems->begin(); it != elems->end(); it++) {
621  if (PushQueue(lock, *it, timeout) == false) {
622  break;
623  }
624 
625  push_num++;
626 
627  /* when get first element, stop waiting */
628  if (timeout >= 0) {
629  timeout = -1;
630  }
631  }
632 
633  /* remove elems from origin sequence */
634  elems->erase(elems->begin(), elems->begin() + push_num);
635 
636  return push_num;
637  }

647  size_t PushQueueForce(std::unique_lock<std::mutex>& lock, Sequence* elems,
648  bool wait_when_full = false, int timeout = 0) {
649  size_t push_num = 0;
650 
651  /* shutdown or queue is not full*/
652  auto wait_check = [&]() {
653  if (wait_when_full == true) {
654  if (capacity_ >= 1 + queue_.size()) {
655  return true;
656  }
657  }
658 
659  return wait_when_full == false || shutdown_ == true || queue_.size() == 0;
660  };
661 
662  /* check if queue has space, or wait. */
663  if (Wait(not_full_, lock, timeout, wait_check) == false) {
664  return false;
665  }
666 
667  for (auto it = elems->begin(); it != elems->end(); it++) {
668  queue_.emplace(*it);
669  push_num++;
670  }
671 
672  /* remove elems from origin sequence */
673  elems->erase(elems->begin(), elems->begin() + push_num);
674 
675  return push_num;

756  bool WaitQueue(std::unique_lock<std::mutex>& lock, int timeout) {
757  /* wait check has data */
758  auto wait_check = [&]() { return shutdown_ == true || queue_.size() > 0; };
759 
760  /* return false when shutdown */
761  if (shutdown_ == true && queue_.size() == 0) {
762  errno = ESHUTDOWN;
763  return false;
764  }
765 
766  /* check if queue has data, or wait. */
767  if (Wait(not_empty_, lock, timeout, wait_check) == false) {
768  return false;
769  }
770 
771  /* shutdown may be called, return false */
772  if (queue_.size() == 0) {
773  errno = ESHUTDOWN;
774  return false;
775  }

857  int timeout = 0, uint32_t max_elems = -1) {
858  size_t num = 0;
859 
860  /* wait for elems first */
861  if (num >= max_elems || _BlockingQueue::WaitQueue(lock, timeout) == false) {
862  return num;
863  }
864 
865  /* Get first element */
866  auto first = _BlockingQueue::queue_.front();
867  auto second = first;
868  num++;
869  elems->emplace_back(std::move(_BlockingQueue::queue_.front()));
870  _BlockingQueue::queue_.pop();
871 
872  /* Loop and get same element */
873  while (_BlockingQueue::queue_.size() > 0 && num < max_elems) {
874  if (_BlockingQueue::WaitQueue(lock, -1) == false) {
875  return num;
876  }
877 
878  second = _BlockingQueue::queue_.front();
879 
880  /* try to get all same priority elements
881  * break when priority is different
882  */
883  if (comp_(first, second) == true || comp_(second, first) == true) {
884  break;
885  }
886 
887  /* get and remove element */
888  elems->emplace_back(std::move(_BlockingQueue::queue_.front()));
889  _BlockingQueue::queue_.pop();
890  num++;
891  }
892

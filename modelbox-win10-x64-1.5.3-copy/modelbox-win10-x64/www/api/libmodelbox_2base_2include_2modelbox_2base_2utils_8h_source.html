139 class RefVar {
140  public:
141  RefVar() {
142  weak_var_ = new std::weak_ptr<T>[max_var_num_];
143  make_func_ = new std::function<std::shared_ptr<T>(int)>[max_var_num_];
144  }
145 
146  RefVar(int max_var_num) {
147  if (max_var_num < 0 || max_var_num > 1024) {
148  max_var_num_ = 0;
149  return;
150  }
151 
152  weak_var_ = new std::weak_ptr<T>[max_var_num];
153  make_func_ = new std::function<std::shared_ptr<T>(int)>[max_var_num];
154  max_var_num_ = max_var_num;
155  }
156 
157  virtual ~RefVar() {
158  delete[] make_func_;
159  delete[] weak_var_;
160  max_var_num_ = 0;
161  }
168  template <class Callable>
169  void MakeFunc(Callable &&make_func, int index = -1) noexcept {
170  if (index < 0) {
171  for (int i = 0; i < max_var_num_; i++) {
172  make_func_[i] = std::forward<Callable>(make_func);
173  }
174  return;
175  }
176 
177  if (index >= max_var_num_) {
178  return;
179  }
180 
181  make_func_[index] = std::forward<Callable>(make_func);
182  }
183 
189  std::vector<std::shared_ptr<T>> GetAll() {
190  std::lock_guard<std::mutex> lock(weak_var_lock_);
191  std::vector<std::shared_ptr<T>> result;
192  for (int i = 0; i < max_var_num_; i++) {
193  const auto &var = weak_var_[i].lock();
194  if (var == nullptr) {
195  continue;
196  }
197 
198  result.emplace_back(var);
199  }
200 
201  return result;
202  }
203 
210  std::shared_ptr<T> Get(int index = 0) {
211  if (index >= max_var_num_) {
212  return nullptr;
213  }
214 
215  auto weak_var = weak_var_[index].lock();
216  if (weak_var) {
217  return weak_var;
218  }
219 
220  std::lock_guard<std::mutex> lock(weak_var_lock_);
221  weak_var = weak_var_[index].lock();
222  if (weak_var) {
223  return weak_var;
224  }
225 
226  if (make_func_[index] == nullptr) {
227  return nullptr;
228  }
229 
230  weak_var = make_func_[index](index);
231  weak_var_[index] = weak_var;
232  return weak_var;
233  }
234 
235  private:
236  std::weak_ptr<T> *weak_var_;
237  std::mutex weak_var_lock_;
238  std::function<std::shared_ptr<T>(int)> *make_func_;
239  int max_var_num_{1};
240 };

180  template <typename func, typename... ts>
181  auto Submit(const std::string &name, func &&fun, ts &&... params)
182  -> std::future<typename std::result_of<func(ts...)>::type> {
183  auto execute =
184  std::bind(std::forward<func>(fun), std::forward<ts>(params)...);
185  using ReturnType = typename std::result_of<func(ts...)>::type;
186  using PackagedTask = std::packaged_task<ReturnType()>;
187  auto package_task = std::make_shared<PackagedTask>(std::move(execute));
188  auto result = package_task->get_future();
189 
190  ThreadFunction task;
191  task.func = [package_task]() { (*package_task)(); };
192  if (name.length() > 0) {
193  task.name = name;
194  }
195 
196  if (SubmitTask(task) == false) {
197  return std::future<typename std::result_of<func(ts...)>::type>();
198  }
199 
200  return result;
201  }

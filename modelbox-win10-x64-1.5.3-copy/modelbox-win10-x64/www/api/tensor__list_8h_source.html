104  template <typename T>
105  Status Build(const std::vector<std::vector<size_t>>& shape_list) {
106  std::vector<size_t> data_size_list(shape_list.size(), 0);
107  std::transform(shape_list.begin(), shape_list.end(), data_size_list.begin(),
108  [](const std::vector<size_t>& shape) {
109  return Volume(shape) * sizeof(T);
110  });
111 
112  size_t size = std::accumulate(data_size_list.begin(), data_size_list.end(),
113  (size_t)0, std::plus<size_t>());
114  if (!bl_->dev_mem_) {
115  return {STATUS_INVALID, "device memory must not be nullptr."};
116  }
117 
118  auto device = bl_->dev_mem_->GetDevice();
119  bl_->dev_mem_ = device->MemAlloc(size);
120  if (!bl_->dev_mem_) {
121  MBLOG_WARN << " MemAlloc " << size << " byte data failed";
122  return STATUS_NOMEM;
123  }
124 
125  bl_->buffer_list_.resize(data_size_list.size(), nullptr);
126 
127  size_t offset = 0;
128  for (size_t i = 0; i < bl_->buffer_list_.size(); i++) {
129  auto&& mem = bl_->dev_mem_->Cut(offset, data_size_list[i]);
130  bl_->buffer_list_[i] = std::make_shared<TensorBuffer>(mem);
131  offset += data_size_list[i];
132  }
133 
134  bl_->is_contiguous_ = true;
135  return SetShape<T>(shape_list);
136  }

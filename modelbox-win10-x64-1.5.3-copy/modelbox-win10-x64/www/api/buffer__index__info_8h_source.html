185 template <typename T>
186 void BufferManageView::GenProcessInfo(
187  const std::unordered_map<std::string, T> &parent_data,
188  size_t data_count_per_port,
189  const std::function<std::shared_ptr<Buffer>(const T &container, size_t idx)>
190  &get_buffer_at,
191  std::vector<std::shared_ptr<BufferProcessInfo>> &process_info_list,
192  bool all_in_one_process) {
193  if (all_in_one_process) {
194  process_info_list.push_back(std::make_shared<BufferProcessInfo>());
195  } else {
196  process_info_list.reserve(data_count_per_port);
197  for (size_t i = 0; i < data_count_per_port; ++i) {
198  process_info_list.push_back(std::make_shared<BufferProcessInfo>());
199  }
200  }
201 
202  for (auto &port_data_item : parent_data) {
203  auto &port_name = port_data_item.first;
204  auto &data_list = port_data_item.second;
205  auto index_info_list =
206  std::make_shared<std::list<std::shared_ptr<BufferIndexInfo>>>();
207  for (size_t i = 0; i < data_count_per_port; ++i) {
208  auto buffer = get_buffer_at(data_list, i);
209  auto index_info = BufferManageView::GetIndexInfo(buffer);
210  index_info_list->push_back(index_info);
211  if (!all_in_one_process) {
212  auto inherit_info = process_info_list[i];
213  inherit_info->SetParentBuffers(port_name, std::move(*index_info_list));
214  index_info_list->clear();
215  }
216  }
217  if (all_in_one_process) {
218  auto inherit_info = process_info_list.front();
219  inherit_info->SetParentBuffers(port_name, std::move(*index_info_list));
220  }
221  }
222 }

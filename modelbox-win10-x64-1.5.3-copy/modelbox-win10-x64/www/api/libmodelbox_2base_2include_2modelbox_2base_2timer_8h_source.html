150 class TimerCompare {
151  public:
152  bool operator()(const std::shared_ptr<TimerTask> &lhs,
153  const std::shared_ptr<TimerTask> &rhs) {
154  auto hit_time_lhs = lhs->GetHitTime();
155  auto hit_time_rhs = rhs->GetHitTime();
156  if (hit_time_lhs == hit_time_rhs) {
157  return lhs->GetDelay() + lhs->GetPeriod() >
158  rhs->GetDelay() + rhs->GetPeriod();
159  }
160 
161  return hit_time_lhs > hit_time_rhs;
162  }
